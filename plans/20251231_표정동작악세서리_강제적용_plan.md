### 방안 1: 표정/동작 강제 추가 함수 구현
**위치**: `services/geminiService.ts`

**함수 1: `ensureDynamicExpression`**
```typescript
const ensureDynamicExpression = (prompt: string, sceneNumber: number): string => {
  // 8가지 다양한 표정 풀
  const expressions = [
    "biting lips nervously",
    "furrowed brows with concern",
    "wide eyes in shock",
    "sweating with tension",
    "flustered and blushing",
    "intense focused gaze",
    "smirking confidently",
    "pouting playfully"
  ];
  
  // 이미 표정이 있는지 확인
  if (/biting|furrowed|wide eyes|sweating|flustered|smirking|pouting|nervous|shocked|confident/i.test(prompt)) {
    return prompt;
  }
  
  // 씬 번호에 따라 다른 표정 강제 추가
  const selectedExpression = expressions[sceneNumber % expressions.length];
  
  // "Korean woman in her 40s," 뒤에 표정 삽입
  return prompt.replace(
    /(Korean woman in her \d+s),/i,
    `$1, ${selectedExpression},`
  );
};
```

**함수 2: `ensureDynamicAction`**
```typescript
const ensureDynamicAction = (prompt: string, sceneNumber: number): string => {
  // 8가지 다양한 동작 풀
  const actions = [
    "leaning forward with curiosity",
    "crossing arms defensively",
    "gesturing expressively with hands",
    "tilting head in confusion",
    "stepping back in surprise",
    "reaching out tentatively",
    "turning away in frustration",
    "laughing while covering mouth"
  ];
  
  // 이미 동작이 있는지 확인
  if (/leaning|crossing|gesturing|tilting|stepping|reaching|turning|laughing|walking|standing/i.test(prompt)) {
    return prompt;
  }
  
  // 씬 번호에 따라 다른 동작 강제 추가
  const selectedAction = actions[sceneNumber % actions.length];
  
  // 표정 뒤에 동작 삽입
  return prompt.replace(
    /(nervous|shocked|confident|blushing|focused|smirking|pouting|concerned),/i,
    `$1, ${selectedAction},`
  );
};
```

### 방안 2: 악세서리 강제 추가 함수 구현

**함수 3: `ensureAccessory`**
```typescript
const ensureAccessory = (prompt: string, characterId: string): string => {
  // Slot별 고정 악세서리
  const accessories = {
    A: "diamond stud earrings + gold luxury watch",
    B: "pearl drop earrings + silver pendant necklace",
    C: "simple platinum hoops + black leather strap watch"
  };
  
  // 이미 악세서리가 있는지 확인
  if (/earrings|necklace|bracelet|watch|ring|accessory/i.test(prompt)) {
    return prompt;
  }
  
  // 캐릭터 ID에서 Slot 추출 (A, B, C)
  const slotMatch = characterId.match(/Slot ([ABC])/i);
  const slot = slotMatch ? slotMatch[1] : 'A';
  
  // Slot에 맞는 악세서리 가져오기
  const accessory = accessories[slot as keyof typeof accessories] || accessories.A;
  
  // "Outfit: [의상]," 뒤에 악세서리 삽입
  return prompt.replace(
    /(Outfit:[^,]+),/i,
    `$1, wearing ${accessory},`
  );
};
```
