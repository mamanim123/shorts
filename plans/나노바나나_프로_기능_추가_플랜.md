// 1. components/PromptGenerator.tsx 생성
const PromptGenerator = () => {
  const [mode, setMode] = useState<'keyword' | 'image'>('keyword');
  const [input, setInput] = useState('');
  const [generatedPrompt, setGeneratedPrompt] = useState('');
  
  const generateFromKeywords = async () => {
    // 키워드를 상세 프롬프트로 확장
    const response = await fetch('/api/expand-prompt', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ keywords: input })
    });
    const { prompt } = await response.json();
    setGeneratedPrompt(prompt);
  };
  
  const extractFromImage = async (imageFile: File) => {
    // 이미지에서 프롬프트 추출
    const formData = new FormData();
    formData.append('image', imageFile);
    
    const response = await fetch('/api/image-to-prompt', {
      method: 'POST',
      body: formData
    });
    const { prompt } = await response.json();
    setGeneratedPrompt(prompt);
  };
  
  return (
    <div className="prompt-generator">
      <div className="mode-selector">
        <button onClick={() => setMode('keyword')}>키워드로 생성</button>
        <button onClick={() => setMode('image')}>이미지에서 추출</button>
      </div>
      
      {mode === 'keyword' ? (
        <div>
          <input 
            type="text" 
            placeholder="키워드 입력 (예: 여성, 빨간 드레스, 골프장)"

            onChange={(e) => setInput(e.target.value)}

          <button onClick={generateFromKeywords}>프롬프트 생성</button>
        </div>
      ) : (
        <input 
          type="file" 
          accept="image/*"
          onChange={(e) => extractFromImage(e.target.files[0])}
        />
      )}
      
      {generatedPrompt && (
        <div className="result">
          <h4>생성된 프롬프트:</h4>
          <textarea value={generatedPrompt} readOnly />
          <button onClick={() => navigator.clipboard.writeText(generatedPrompt)}>
            복사
          </button>
        </div>
      )}
    </div>
  );
};

// 1. components/SequentialImageGenerator.tsx
interface SequentialConfig {
  baseImage: File | null;
  basePrompt: string;
  variations: string[]; // 각 프레임별 변형 프롬프트
  frameCount: number;
  fps: number;
}

const SequentialImageGenerator = () => {
  const [config, setConfig] = useState<SequentialConfig>({
    baseImage: null,
    basePrompt: '',
    variations: [],
    frameCount: 10,
    fps: 24
  });
  const [generatedFrames, setGeneratedFrames] = useState<string[]>([]);
  
  const generateSequence = async () => {
    const frames = [];
    
    for (let i = 0; i < config.frameCount; i++) {
      const framePrompt = `${config.basePrompt}, ${config.variations[i] || ''}`;
      
      const response = await fetch('/api/generate-frame', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          prompt: framePrompt,
          referenceImage: config.baseImage ? await toBase64(config.baseImage) : null,
          frameNumber: i
        })
      });
      
      const { imageUrl } = await response.json();
      frames.push(imageUrl);
      setGeneratedFrames([...frames]);
    }
  };
  
  return (
    <div className="sequential-generator">
      <h2>연속 이미지 생성</h2>
      
      <div className="config">
        <input 
          type="file" 
          accept="image/*"
          onChange={(e) => setConfig({...config, baseImage: e.target.files[0]})}
        />
        <input 
          type="text"
          placeholder="기본 프롬프트"
          value={config.basePrompt}
          onChange={(e) => setConfig({...config, basePrompt: e.target.value})}
        />
        <input 
          type="number"
          placeholder="프레임 수"
          value={config.frameCount}
          onChange={(e) => setConfig({...config, frameCount: parseInt(e.target.value)})}
        />
        <input 
          type="number"
          placeholder="FPS"
          value={config.fps}
          onChange={(e) => setConfig({...config, fps: parseInt(e.target.value)})}
        />
        
        <h3>프레임별 변형</h3>
        {Array.from({length: config.frameCount}).map((_, i) => (
          <input 
            key={i}
            type="text"
            placeholder={`프레임 ${i+1} 변형 (선택사항)`}
            onChange={(e) => {
              const newVariations = [...config.variations];
              newVariations[i] = e.target.value;
              setConfig({...config, variations: newVariations});
            }}
          />
        ))}
        
        <button onClick={generateSequence}>시퀀스 생성</button>
      </div>
      
      <div className="frames-preview">
        {generatedFrames.map((frame, i) => (
          <img key={i} src={frame} alt={`Frame ${i+1}`} />
        ))}
      </div>
      
      {generatedFrames.length === config.frameCount && (
        <button onClick={() => createVideo(generatedFrames, config.fps)}>
          비디오로 변환
        </button>
      )}
    </div>
  );
};

// 1. components/PromptLibrary.tsx
interface SavedPrompt {
  id: string;
  name: string;
  prompt: string;
  category: string;
  tags: string[];
  createdAt: string;
}

const PromptLibrary = () => {
  const [prompts, setPrompts] = useState<SavedPrompt[]>([]);
  const [filter, setFilter] = useState<string>('');
  
  const savePrompt = (name: string, prompt: string, category: string, tags: string[]) => {
    const newPrompt: SavedPrompt = {
      id: Date.now().toString(),
      name,
      prompt,
      category,
      tags,
      createdAt: new Date().toISOString()
    };
    
    const updated = [...prompts, newPrompt];
    setPrompts(updated);
    localStorage.setItem('promptLibrary', JSON.stringify(updated));
  };
  
  const loadPrompt = (id: string) => {
    const prompt = prompts.find(p => p.id === id);
    if (prompt) {
      // 프롬프트를 생성기에 적용
      navigator.clipboard.writeText(prompt.prompt);
      alert('프롬프트가 클립보드에 복사되었습니다!');
    }
  };
  
  const deletePrompt = (id: string) => {
    const updated = prompts.filter(p => p.id !== id);
    setPrompts(updated);
    localStorage.setItem('promptLibrary', JSON.stringify(updated));
  };
  
  const filteredPrompts = prompts.filter(p => 
    p.name.includes(filter) || 
    p.prompt.includes(filter) || 
    p.tags.some(tag => tag.includes(filter))
  );
  
  return (
    <div className="prompt-library">
      <h2>프롬프트 라이브러리</h2>
      
      <input 
        type="text"
        placeholder="검색..."
        value={filter}
        onChange={(e) => setFilter(e.target.value)}
      />
      
      <div className="prompt-list">
        {filteredPrompts.map(prompt => (
          <div key={prompt.id} className="prompt-item">
            <h3>{prompt.name}</h3>
            <p className="category">{prompt.category}</p>
            <p className="prompt-text">{prompt.prompt}</p>
            <div className="tags">
              {prompt.tags.map(tag => <span key={tag} className="tag">{tag}</span>)}
            </div>
            <div className="actions">
              <button onClick={() => loadPrompt(prompt.id)}>사용</button>
              <button onClick={() => deletePrompt(prompt.id)}>삭제</button>
            </div>
          </div>
        ))}
      </div>
      
      <div className="add-prompt">
        <h3>새 프롬프트 저장</h3>
        {/* 입력 폼 */}
      </div>
    </div>
  );
